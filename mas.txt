<label>         ::= <ident>:
<addr>          ::= <int>|=<ident>
<cmp-op>        ::= "<" | ">" | "<=" | ">=" | =
<calc-op>       ::= + | - | * | / | % | < | >
<range>         ::= <|lb:int>..<|hb:int> | <int>
<reg>           ::= L0|L1|L2|L3
<alias>         ::= <ident>=<int>

<raw-command>   ::= cmd <string>
<move>          ::= mov <dst:reg> <src:reg>
<set>           ::= set <reg> <int>
<load>          ::= load <addr>
<store>         ::= store <addr>
<cmp>           ::= cmp <cmp-op>
<cmp-in>        ::= cmpin <range>
<branch>        ::= b <label:ident>
<branch-if>     ::= bi <label:ident>
<branch-if-not> ::= bn <label:ident>
<calculate>     ::= calc <compute-op>
<random>        ::= rand <min:int> <max:int>
<call>          ::= call <int> <label>
<debug>         ::= debug <string>
<log>           ::= log <string>
