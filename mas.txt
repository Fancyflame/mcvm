<label>         ::= <ident>:
<addr>          ::= <int>|=<ident>
<operator>      ::= "<" | ">" | "<=" | ">=" | =
<range>         ::= <|lb:int>..<|hb:int>
<reg>           ::= L0|L1|L2|L3
<alias>         ::= <ident>=<int>

<raw-command>   ::= cmd <string>
<copy>          ::= copy <dst:reg> <src:reg>
<set>           ::= set <reg> <int>
<load>          ::= load <reg> <addr>
<store>         ::= store <reg> <addr>
<cmp>           ::= cmp <operator>
<cmp-in>        ::= cmpin <range>
<branch>        ::= b <label:ident>
<branch-if>     ::= bi <label:ident>
<branch-if-not> ::= bn <label:ident>
<add>           ::= add
<call>          ::= call <int> <label>
